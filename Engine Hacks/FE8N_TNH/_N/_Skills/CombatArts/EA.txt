
#define WAR_CONFIG (0x0203AF00)
#define ARROW_CONFIG (WAR_CONFIG+0x7)

#include list_definitions.event

//////////////////

PUSH
ORG 0x02763c
 jumpToHack(CombatSelector)

POP

CombatSelector:
	#incbin Arrow.dmp
	WORD ARROW_CONFIG
ALIGN 4

////////(help_beta)
PUSH
 ORG 0x08b5ac
 jumpToHack(combat_help)
POP
ALIGN 4
combat_help:
#incbin help.dmp
  WORD WAR_CONFIG
  WORD ARROW_CONFIG
  POIN skl_table
  WORD ICON_LIST_SIZE
  WORD 0
  WORD 0x0203E775//0203E764(アイテムヘルプのベース)

/////////////////////////////////////
PUSH
	ORG 0x01e684
	jumpToHack(grafix_ws)
POP
grafix_ws:
	#incbin grafix.dmp
	WORD WAR_CONFIG
	WORD ARROW_CONFIG
	POIN skl_table
	WORD ICON_LIST_SIZE
	POIN getSkill
	POIN combat_art_table
	WORD COMBAT_ART_LIST_SIZE
	POIN mergeOccult
	
ALIGN 4


/////////////revert
PUSH
ORG 0x022c60
SHORT $b500 $f7fa $fd65 $2000
POP

////////////////
PUSH
ORG 0x022828
	jumpToHack(cancelWarSkill)
POP
cancelWarSkill:
	#incbin cancelWarSkill.dmp
	WORD WAR_CONFIG
	ALIGN 4

////////////////////////
PUSH
ORG 0x050008
	jumpToHack(WarSkillSelector)
POP
WarSkillSelector:
	#incbin Selector.dmp
	WORD ARROW_CONFIG
	WORD WAR_CONFIG
	ALIGN 4

////////////////////攻撃後処理
PUSH
	ORG 0x02b866
	SHORT $46C0
	SHORT $2000
//統合
	SHORT $46C0
	jumpToHack(afterAttack)

//以下エフェクト関連
//必的を貫通エフェクトに変更
ORG 0x02b174
	BYTE $6D $02

//必的の前エフェクト削除
ORG 0x059e44
	BYTE $12 $E0

//必的でも貫通ヒット音ON
ORG 0x074318
	BYTE $00 $05 $01 $D4 $80 $00 $0C $D5
ORG 0x074384
	BYTE $00 $05 $01 $D4 $80 $00 $0C $D5
ORG 0x0743F0
	BYTE $00 $05 $01 $D4 $80 $00 $0C $D5
ORG 0x07496a
	BYTE $80 $21
	BYTE $00 $05 $01 $D4 $80 $00 $0B $D5

//ORG 0x06ebbc
//	BYTE $80 $25 $AD $00 $05 $40 $02 $D1 $00 $05 $04 $D5 $C0 $46

POP

afterAttack:
	#incbin afterAttack.dmp
	POIN InfiniteItem
	POIN hasRagingStorm
	POIN combat_art_table
	WORD COMBAT_ART_LIST_SIZE
	POIN HasLunge
	POIN HasHitAndRun
	POIN HasKnockBack
	ALIGN 4

//////////////////////////
PUSH

ORG 0x2a2bc //setFlag
BYTE	$00 $49 $8F $46
POIN	nagi_setFlag

POP


nagi_setFlag:
  #incbin setFlag.dmp
	WORD WAR_CONFIG
	WORD ARROW_CONFIG
	ALIGN 4
